Quick Start

Get started in minutes

Installation

// Include CAMELTOW in your project
<script src="cameltow.min.js"></script>

// Or install via npm
npm install cameltow

Basic Usage

// Initialize CAMELTOW
const camelTow = new CAMELTOW({
    secret: 'your-secret-key',
    algorithm: 'HS256'
});

// Generate a token
const token = camelTow.generate({
    userId: 'user123',
    expiresIn: '24h',
    scope: 'read write'
});

// Validate a token
const payload = camelTow.verify(token);
if (payload.isValid) {
    console.log('Token is valid!', payload.data);
}
